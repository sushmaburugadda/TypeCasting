public class Arrays {
	public static void main(String[] args) {

		//Single-dimensional array
		int a[]= {1,2,3,4,5};
		for(int i=0;i<5;i++) {
		System.out.println("Elements of array a: "+a[i]);
		}


		//multidimensional array
		int b[][]  = {
		            {1,3,5,6,8}, 
		            {2,4,6,8} };
		      
		      System.out.println("\nLength of row 1: " + b[1].length);
		      }
		}
